def Least_moves(x):
    moves=0
    while x>1:
         if x%2==0:
             x=x/2
             moves=moves+1
         else:
             x=x-1
             moves=moves+1

def Find_expression(value, i=0): # 用于找到表达式和表达式数量，并打印出来。
